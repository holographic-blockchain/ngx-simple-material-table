Basic Table with Pagination

Pagination is supported with the MatPaginator component. It can be placed inside the hb-mat-table component to have it automatically hooked up. This requires the use of a MatTableDataSource. The paginator will be placed immediately below the table. For more information on the MatPaginator please see https://material.angular.io/components/paginator/overview. 

<hb-mat-table [tableData]="matGridData">
    <hb-mat-table-column name="firstName"></hb-mat-table-column>
    <hb-mat-table-column name="lastName"></hb-mat-table-column>
    <hb-mat-table-column name="age" dataType="number"></hb-mat-table-column>
    <hb-mat-table-column name="birthDate" dataType="date"></hb-mat-table-column>
    <mat-paginator [pageSizeOptions]="[5, 10, 25]"></mat-paginator>
</hb-mat-table>

tableData: UserInfo[] = ...;
matGridData: MatTableDataSource<UserInfo>;

constructor() {{ '{' }}
    this.matGridData = new MatTableDataSource<UserInfo>(this.tableData);
}

class UserInfo {{ '{' }}
    userId: number | null = null;
    firstName: string | null = null;
    lastName: string | null = null;
    age: number | null = null;
    birthDate: Date | null = null;
}
Basic Table, Pagination Handled Outside of Component

For more control over the layout of the paginator, it can be added outside of the hb-mat-table component and attached manually. In the below example, the paginator is placed above the table. 

<mat-paginator #paginator
               [pageSizeOptions]="[5, 10, 25]"></mat-paginator>
<hb-mat-table [tableData]="matGridData">
    <hb-mat-table-column name="firstName"></hb-mat-table-column>
    <hb-mat-table-column name="lastName"></hb-mat-table-column>
    <hb-mat-table-column name="age" dataType="number"></hb-mat-table-column>
    <hb-mat-table-column name="birthDate" dataType="date"></hb-mat-table-column>
</hb-mat-table>

tableData: UserInfo[] = ...;
matGridData: MatTableDataSource<UserInfo>;

@ViewChild('paginator') matPaginator!: MatPaginator;

constructor() {{ '{' }}
    this.matGridData = new MatTableDataSource<UserInfo>(this.tableData);
}

ngAfterViewInit(): void {{ '{' }}
    this.matGridData.paginator = this.matPaginator;
}

class UserInfo {{ '{' }}
    userId: number | null = null;
    firstName: string | null = null;
    lastName: string | null = null;
    age: number | null = null;
    birthDate: Date | null = null;
}
