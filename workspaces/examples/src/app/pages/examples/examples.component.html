Basic Table Displaying Object Properties

Binds to an array of objects, displaying the object property values as-is. For best results use a MatTableDataSource which includes default sorting and pagination(???). 

<hb-mat-table [tableData]="matGridData">
    <hb-mat-table-column name="string1"></hb-mat-table-column>
    <hb-mat-table-column name="string2"></hb-mat-table-column>
    <hb-mat-table-column name="stringWithALongName"></hb-mat-table-column>
    <hb-mat-table-column name="number1" dataType="number"></hb-mat-table-column>
    <hb-mat-table-column name="dateDate" dataType="date"></hb-mat-table-column>
    <hb-mat-table-column name="dateString" dataType="date"></hb-mat-table-column>
</hb-mat-table>

tableData: DataEntity[] = ...;
matGridData: MatTableDataSource<DataEntity>;

constructor() {{ '{' }}
    this.matGridData = new MatTableDataSource<DataEntity>(this.tableData);
}

class DataEntity {{ '{' }}
    string1: string | null = null;
    string2: string | null = null;
    stringWithALongName: string | null = null;
    number1: number | null = null;
    dateDate: Date | number | null = null;
    dateString: string | null = null;
}
